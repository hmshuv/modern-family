//FIFO ASG8
#include <stdio.h>
void fifoPageReplacement(int pages[], int n, int frames) {
 int frameArray[frames];
 int pageFaults = 0;
 int index = 0;
 // Initialize frame array with -1
 for (int i = 0; i < frames; i++) {
 frameArray[i] = -1;
 }
 printf("\nSteps for page replacement:\n");
 for (int i = 0; i < n; i++) {
 int page = pages[i];
 int found = 0; // Page hit flag
 // Check if page is already in frame
 for (int j = 0; j < frames; j++) {
 if (frameArray[j] == page) {
 found = 1;
 break;
 }
 }
 // If not found -> Page fault (miss)
 if (!found) {
 frameArray[index] = page;
 index = (index + 1) % frames;
 pageFaults++;
 }
 // Print frame contents after each reference
 printf("Page %d -> Frames: [", page);
 for (int j = 0; j < frames; j++) {
 if (frameArray[j] != -1)
 printf("%d", frameArray[j]);
 else
 printf("-");
 if (j < frames - 1) printf(" ");
 }
 printf("]");
 if (found)
 printf(" => HIT\n");
 else
 printf(" => MISS\n");
 }
 printf("\nTotal Page Faults: %d\n", pageFaults);
}
int main() {
 int n, frames;
 char continueChoice;
 do {
 printf("Enter the number of pages in reference string: ");
 scanf("%d", &n);
 int pages[n];
 printf("Enter the pages in the reference string:\n");
 for (int i = 0; i < n; i++) {
 scanf("%d", &pages[i]);
 }
 printf("Enter the number of frames: ");
 scanf("%d", &frames);
 fifoPageReplacement(pages, n, frames);
 printf("\nDo you want to perform another paging operation? (y/n): ");
 scanf(" %c", &continueChoice);
 } while (continueChoice == 'y' || continueChoice == 'Y');
 return 0;
}
