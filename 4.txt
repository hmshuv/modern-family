//FCFS ASG3
#include <stdio.h> 
 
int main() { 
int arrival[3], burst[3], completion[3], turnaround[3], waiting[3]; 
int pid[3] = {1, 2, 3}; 
 int i, j; 
float Total_WT = 0.0, Total_TAT = 0.0; 
int n; 
 printf("Enter the no of processes"); 
 scanf(" %d",&n); 
 
 for(i = 0; i < n; i++) { 
 printf("Enter Arrival Time and Burst Time for Process %d: ", i + 1); 
 scanf("%d %d", &arrival[i], &burst[i]); 
 } 
 
 for(i = 0; i < n-1; i++) { 
 for(j = 0; j < n-1 - i; j++) { 
if(arrival[j] > arrival[j+1]) { 
 
 int temp = arrival[j]; 
 arrival[j] = arrival[j+1]; 
 arrival[j+1] = temp; 
 
 temp = burst[j]; 
 burst[j] = burst[j+1]; 
 burst[j+1] = temp; 
 
 temp = pid[j]; 
 pid[j] = pid[j+1]; 
 pid[j+1] = temp; 
 } 
 } 
 } 
 
int current_time = 0; 
for(i = 0; i < n; i++) { 
if(current_time < arrival[i]) { 
 current_time = arrival[i]; 
 } 
current_time += burst[i]; 
 completion[i] = current_time; 
 waiting[i] = turnaround[i] - burst[i]; 
 } 
 
 printf("\nProcess\tAT\tBT\tCT\tTAT\tWT\n"); 
 for(i = 0; i < n; i++) { 
 printf("P%d\t%d\t%d\t%d\t%d\t%d\n", pid[i], arrival[i], burst[i], completion[i], turnaround[i], waiting[i]); 
 } 
 
 printf("\nGantt Chart:\n|"); 
for(i = 0; i < n; i++) { 
 printf("P%d |", pid[i]); 
 } 
 printf("\n0"); 
for(i = 0; i < n; i++) { 
 printf("%d", completion[i]); 
 } 
 printf("\n"); 
 
 for(int i = 0; i < n; i++) { 
 Total_WT += waiting[i]; 
 Total_TAT += turnaround[i]; 
 } 
 
 printf("\nAverage Waiting Time: %f\n", Total_WT / n); 
 printf("\nAverage Turnaround Time: %f\n", Total_TAT / n); 
 
 return 0; 
}
